JavaScript Operators

Example 1: Arithmetic Operators

// Arithmetic operators: +, -, *, /, %, ++, --
let a = 10, b = 3;

let addition = a + b;
let subtraction = a - b;
let multiplication = a * b;
let division = a / b;
let remainder = a % b;
let increment = ++a;  // Pre-increment
let decrement = --b;  // Pre-decrement

console.log(addition, subtraction, multiplication, division, remainder, increment, decrement);

Example 2: Comparison Operators

// Comparison operators: ==, ===, !=, !==, >, <, >=, <=
let x = 5, y = '5';

let equality = (x == y);      // Loose equality (checks value only)
let strictEquality = (x === y);  // Strict equality (checks value and type)
let notEqual = (x != y);      // Loose inequality
let strictNotEqual = (x !== y); // Strict inequality
let greaterThan = (x > 3);    // Greater than
let lessThan = (x < 10);      // Less than
let greaterThanEqual = (x >= 5); // Greater than or equal to
let lessThanEqual = (x <= 5); // Less than or equal to

console.log(equality, strictEquality, notEqual, strictNotEqual, greaterThan, lessThan, greaterThanEqual, lessThanEqual);

Example 3: Logical Operators

// Logical operators: && (AND), || (OR), ! (NOT)
let isTrue = true, isFalse = false;

let andOperator = (isTrue && isFalse);  // AND operator
let orOperator = (isTrue || isFalse);   // OR operator
let notOperator = !isTrue;              // NOT operator

console.log(andOperator, orOperator, notOperator);

Example 4: Assignment Operators

// Assignment operators: =, +=, -=, *=, /=, %=
let num = 10;

num += 5;  // num = num + 5
num -= 3;  // num = num - 3
num *= 2;  // num = num * 2
num /= 4;  // num = num / 4
num %= 3;  // num = num % 3

console.log(num);
